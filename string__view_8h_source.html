70 constexpr string_view substr(int32_t start, size_t count = npos) const {
71 // This is different behavior than the standard, since this doesn't
72 // throw if out of bounds. Instead, it silently crops to the
73 // bounds. Clearly will also have unwanted behavior if `start` is a
74 // negative signed int.
75 if (unlikely(start < 0)) {
