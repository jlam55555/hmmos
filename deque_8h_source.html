227 // Note we store a lot of state in this iterator.
228 //
229 // - This can _not_ be invalidated on \a deque::erase() if erasing
230 // the first or last element, hence we store the index with \a
231 // start_idx factored in.
232 // - This must support a wrapped block array, so we need to store
233 // \a blocks, \a block_count, and \a start_block. Just \a
234 // start_block won't suffice.
235 // - This shouldn't be invalidated on deque::swap(), so we can't
236 // keep a reference to the deque that contains it.
237 // - This can be invalidated on (any) insertion, notably if \a
238 // start_block, \a block_count, and/or \a blocks changes.
239 //
240 // Luckily, the prefix increment/decrement/comparison operators
241 // are fairly lightweight, since they assume that the other
242 // members (which are only used for dereferencing) are constant
243 // for non-iterator-invalidating operations.
244 //
245 // I'd like for these fields to be const, but this iterator must
246 // satisfy \a weakly_incrementable which requires movability. That
247 // only happens on post-increment, which is always less
248 // performant.
249 U **blocks;
250 size_t block_count;
251 size_t start_block;
252 size_t idx;

311template <typename T> void deque<T>::shrink_to_fit() {
312 // Deallocate extra blocks at the end.
313 for (size_t i =
314 (start_block + ((start_idx + _size - 1) / elems_per_block + 1)) %
315 _block_count;
316 i != start_block && _blocks[i] != nullptr; i = (i + 1) % _block_count) {
317 BlockAllocator{}.deallocate(
318 (typename BlockAllocator::value_type *)_blocks[i], 1);
319 _blocks[i] = nullptr;
320 }
321 // Deallocate extra blocks at start.
322 for (; start_idx >= elems_per_block;
323 start_idx -= elems_per_block,
324 start_block = (start_block + 1) % _block_count) {
325 BlockAllocator{}.deallocate(
326 (typename BlockAllocator::value_type *)_blocks[start_block], 1);
327 _blocks[start_block] = nullptr;
328 }
329 // The above cases don't account for an empty deque. There's
330 // probably a cleaner way to do this but whatevs.
331 if (empty() && _blocks[start_block] != nullptr) {
332 start_idx = 0;
333 BlockAllocator{}.deallocate(
334 (typename BlockAllocator::value_type *)_blocks[start_block], 1);
335 _blocks[start_block] = nullptr;
336 }
337}

339template <typename T>
340auto deque<T>::insert(const_iterator pos, size_t count, const T &value)
341 -> iterator {
342 const size_t idx = pos.idx - start_idx;
343 size_t cur_idx = idx;
344 // Note: this fails if inserting count>1 and we're not inserting
345 // at the back, since we don't currently support inserting in the
346 // middle.
347 for (size_t i = 0; i < count; ++i) {
348 // Iterator is invalidated after insert, so we have to keep
349 // re-creating it.
350 insert(pos, value);
351 pos = make_iterator(++cur_idx);
352 }
353 return make_iterator(idx);
354}

372template <typename T>
373template <typename... Args>
374auto deque<T>::emplace(const_iterator pos, Args &&...args) -> iterator {
375 const size_t idx = pos.idx - start_idx;
376 size_t i = 0;
377 if (pos == end()) {
378 if ((start_idx + size()) % elems_per_block == 0) {
379 allocate_block(/*start=*/false);
380 }
381 i = size();
382 } else if (pos == begin()) {
383 if (start_idx == 0) {
384 allocate_block(/*start=*/true);
385 start_idx = elems_per_block;
386 }
387 --start_idx;
388 i = 0;
389 } else {
390 // TODO: support insertions in the middle. This makes my life
391 // easier and insertions in the middle are inefficient anyways.
392 ASSERT(false);
393 }
394 ::new (std::addressof((*this)[i])) T(std::forward<Args>(args)...);
395 ++_size;
396 return make_iterator(idx);
397}

400 const size_t idx = pos.idx - start_idx;
401 (*this)[idx].~T();
402 if (pos == begin()) {
403 ++start_idx;
404 } else if (pos == std::prev(end())) {
405 // pass
406 } else {
407 // TODO: support erasing from the middle
408 ASSERT(false);
409 }
410 --_size;
411 return make_iterator(idx);

418 if (it == last) {
419 return it;

458template <typename T> void deque<T>::allocate_block(bool start) {
459 const auto realloc_blocks_array = [&] {
460 size_t new_block_count = _block_count * 2;
461 T **new_blocks = InternalAllocator{}.allocate(new_block_count);
462
463 // Copy _blocks[start_block, start_block+_block_count)
464 // (potentially wrapped) to new_blocks[0, _block_count) (not
465 // wrapped).
466 for (size_t from = start_block, to = 0; to < _block_count;
467 ++to, from = (from + 1) % _block_count) {
468 ASSERT(_blocks[from] != nullptr);
469 new_blocks[to] = _blocks[from];
470 }
471 std::fill(new_blocks + _block_count, new_blocks + new_block_count, nullptr);
472
473 InternalAllocator{}.deallocate(_blocks, _block_count);
474 _block_count = new_block_count;
475 _blocks = new_blocks;
476 start_block = 0;
477 };
478
479 if (start) {
480 if (!start_idx && size() > elems_per_block * (_block_count - 1)) {
481 // All blocks full, reallocate blocks array.
482 realloc_blocks_array();
483 }
484
485 if (!start_idx) {
486 // Allocate a new block.
487 start_block = (start_block + _block_count - 1) % _block_count;
488 _blocks[start_block] = (T *)BlockAllocator{}.allocate(1);
489 }
490 } else {
491 const size_t end_idx = start_idx + size();
492 if (end_idx >= elems_per_block * _block_count) {
493 // All blocks full, reallocate blocks array.
494 realloc_blocks_array();
495 }
496
497 const auto [block_idx, elem_idx] =
498 get_block_elem_idx(end_idx, start_block, _block_count);
499 T *&block = _blocks[block_idx];
500 if (elem_idx == 0 && block == nullptr) {
501 // Allocate a new block.
502 block = (T *)BlockAllocator{}.allocate(1);
503 }
504 }
505}
